#include <stdio.h>
int main(){
    int a,b;
    scanf("%d %d",&a,&b);
    int matrix[a][b];
    for (int i=0;i<a;i++)
    {
        for (int j=0;j<b;j++)
        {
            scanf("%d",&matrix[i][j]);
        }
    }
    int sum=0;
    for (int i=0;i<a;i++)
    {
        for (int j=0;j<b;j++)
        {
            if (i==0||i==a-1||j==0||j==b-1)
            {
                sum+=matrix[i][j];
            }
        }
    }
    printf("%d\n",sum);
    return 0;
}
