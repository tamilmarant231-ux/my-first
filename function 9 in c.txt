#include <stdio.h>
int main()
{
    int N,original,rev=0,digit;
    scanf("%d",&N);
    original=N;  
    while (N>0)
    {
        digit=N%10;
        rev=rev*10+digit;
        N/=10;
    }
    if (rev==original)
        printf("Palindrome\n");
    else
        printf("Not Palindrome\n");
    return 0;
}
