#include <stdio.h>
int main(){
    int a,b;
    scanf("%d %d",&a,&b);
    int c[a][b];
    for(int i=0;i<a;i++)
        for(int j=0;j<b;j++)
            scanf("%d",&c[i][j]);
    int max=0,row=0;
    for(int i=0;i<a;i++)
    {
        int count=1;
        for(int j=0;j<b-1;j++)
        {
            if((c[i][j]%2)!=(c[i][j+1]%2))
                count++;
            else
                break;
        }
        if(count>max)
        {
            max=count;
            row=i;
        }
    }
    printf("%d",row);
    return 0;
}
