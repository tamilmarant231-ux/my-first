#include <stdio.h>
void printFibonacci(int N) {
    int t1=0,t2=1,next;
    for (int i=1;i<=N;i++)
    {
        printf("%d ",t1);
        next=t1+t2;
        t1=t2;
        t2=next;
    }
    printf("\n");
}
int main() {
    int N;
    scanf("%d",&N);
    printFibonacci(N);
    return 0;
}
