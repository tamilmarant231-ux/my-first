#include <stdio.h>
int main()
{
    int a,b,c[50][50],i,j,sum,count=0;
    scanf("%d %d",&a,&b);
    for(i=0;i<a;i++)
        for(j=0;j<b;j++)
            scanf("%d",&c[i][j]);
    for(i=0;i<a;i++)
    {
        sum=0;
        for(j =0;j<b;j++)
        {
            if(j!=i)
                sum+=c[i][j];
        }
        if(c[i][i]>sum)
            count++;
    }
    printf("%d",count);
    return 0;
}
