#include <stdio.h>
#include <stdlib.h>
int main(){
    int a,b;
    scanf("%d %d",&a,&b);
    int matrix[a][b];
    for(int i=0;i<a;i++)
    {
        for(int j=0;j<b;j++)
        {
            scanf("%d",&matrix[i][j]);
        }
    }
    int maxDiff=0;
    for(int i=0;i<a;i++)
    {
        for(int j=0;j<b;j++)
        {
            for(int k=0;k<a;k++)
            {
                if(i!=k)
                {
                    for(int l=0;l<b;l++)
                    {
                        int diff=abs(matrix[i][j]-matrix[k][l]);
                        if(diff>maxDiff)
                        {
                            maxDiff=diff;
                        }
                    }
                }
            }
        }
    }
    printf("%d",maxDiff);
    return 0;
}
