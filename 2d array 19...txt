#include <stdio.h>
int main() {
    int a,b;
    scanf("%d %d",&a,&b);
    int matrix[a][b];
    for(int i=0;i<a;i++)
    {
        for(int j=0;j<b;j++)
        {
            scanf("%d",&matrix[i][j]);
        }
    }
    int count=0;
    for(int i=0;i<a;i++)
    {
        int unique=1;
        for(int j=0;j<b;j++)
        {
            for(int k=j+1;k<b;k++)
            {
                if(matrix[i][j]==matrix[i][k])
                {
                    unique=0;
                    break;
                }
            }
            if(unique==0)
                break;
        }
        if(unique)
            count++;
    }
    printf("%d",count);
    return 0;
}
