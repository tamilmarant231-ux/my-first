#include <stdio.h>

int main(){
    int a,b;
    scanf("%d %d",&a,&b);
    int matrix[a][b];
    for (int i=0;i<a;i++)
    {
        for (int j=0;j<b;j++)
        {
            scanf("%d",&matrix[i][j]);
        }
    }
    int count=0;
    for (int i=0;i<a;i++)
    {
        int strictly_increasing=1; 
        for (int j=1;j<b;j++){
            if (matrix[i][j]<=matrix[i][j-1])
            {
                strictly_increasing=0; 
                break;
            }
        }
        if (strictly_increasing)
            count++;
    }
    printf("%d\n",count);
    return 0;
}
