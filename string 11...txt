#include <stdio.h>
#include <ctype.h>
int main(){
    char id[100];
    int count=0;
    int i=0;
    scanf("%s",id);
    while (id[i]!='\0')
    {
        if (isdigit(id[i]))
        {
            count++;
        }
        i++;
    }
    printf("Number of digits: %d\n",count);
    return 0;
}
