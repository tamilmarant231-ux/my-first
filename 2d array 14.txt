#include <stdio.h>

int main() {
    int a,b;
    scanf("%d %d",&a,&b);
    int matrix[a][b];
    for(int i=0;i<a;i++)
        for(int j=0;j<b;j++)
            scanf("%d",&matrix[i][j]);
    int maxCol=0;
    int maxSum=-1000000;
    for(int j=0;j<b;j++)
    {
        int sum=0;
        for(int i=0;i<a;i++)
            sum+=matrix[i][j];
        if(sum > maxSum)
        {
            maxSum=sum;
            maxCol=j;
        }
    }
    printf("%d", maxCol);
    return 0;
}
