#include <stdio.h>
int main() {
    int a,b;
    scanf("%d %d",&a,&b);
    int matrix[a][b];
    for(int i=0;i<a;i++)
        for(int j=0;j<b;j++)
            scanf("%d",&matrix[i][j]);
    int count=0;
    for(int i=0;i<a;i++)
    {
        int isPalindrome=1;
        for(int j=0;j<b/2;j++)
        {
            if(matrix[i][j]!=matrix[i][b-j-1])
            {
                isPalindrome=0;
                break;
            }
        }
        if(isPalindrome)
            count++;
    }
    printf("%d",count);
    return 0;
}
