#include <stdio.h>
int main(){
    int a,b;
    scanf("%d %d",&a,&b);
    int c[a][b];
    for(int i=0;i<a;i++)
        for(int j=0;j<b;j++)
            scanf("%d",&c[i][j]);
    for(int i=0;i<a;i++)
    {
        for(int j=0;j<b;j++)
        {
            int count=0;
            for(int k=0;k<a;k++)
            {
                for(int l=0;l<b;l++)
                {
                    if(c[i][j]==c[k][l])
                        count++;
                }
            }
            if(count==1)
            {
                printf("%d",c[i][j]);
                return 0;
            }
        }
    }
    return 0;
}
