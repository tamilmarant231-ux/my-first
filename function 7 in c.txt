#include <stdio.h>
int isPrime(int N)
{
    if (N<=1)
    return 0;
    for(int i=2;i<N;i++)
    {
        if (N%i==0)
        return 0; 
    }
    return 1; 
}
int main()
{
    int N;
    scanf("%d",&N);
    if (isPrime(N))
        printf("Prime\n");
    else
        printf("Not Prime\n");
    return 0;
}
