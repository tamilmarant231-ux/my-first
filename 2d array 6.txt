#include <stdio.h>
#include <stdlib.h>

int main() {
    int R, C;
    scanf("%d %d",&R,&C);

    int a[100][100], sum[100];

    for(int i=0;i<R;i++){
        sum[i]=0;
        for(int j=0;j<C;j++){
            scanf("%d",&a[i][j]);
            sum[i]+=a[i][j];
        }
    }

    int max=0;

    for(int i=0;i<R;i++){
        for(int j=i+1;j<R;j++){
            int diff=abs(sum[i]-sum[j]);
            if(diff>max)
                max=diff;
        }
    }

    printf("%d",max);

    return 0;
}
