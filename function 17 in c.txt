#include <stdio.h>
int main() {
    int N,original,sum=0,digit,temp,i;
    scanf("%d",&N);
    original=N;
    int digits=0;
    temp=N;
    while (temp>0)
    {
        temp/=10;
        digits++;
    }
    temp=N;
    while (temp>0)
    {
        digit=temp%10;
        int power=1;
        for (i=1;i<=digits;i++)
            power *= digit;
        sum+=power;
        temp/=10;
    }
    if (sum==original)
        printf("Armstrong\n");
    else
        printf("Not Armstrong\n");
    return 0;
}
