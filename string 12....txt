#include <stdio.h>
#include <ctype.h>
int main(){
    char str[100];
    int i=0;
    int count=0;
    scanf("%s",&str);
    while (str[i]!='\0')
    {
        if (!isalnum(str[i]))
        {
            count++;
        }
        i++;
    }
    printf("special characters:%d\n",count);
    return 0;
}
