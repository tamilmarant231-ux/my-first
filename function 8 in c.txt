#include <stdio.h>
int reverseNumber(int N)
{
    int rev=0;
    while (N>0)
    {
        rev=rev*10+(N%10);  
        N/=10;                    
    }
    return rev;
}
int main()
{
    int N;
    scanf("%d",&N);
    int reversed=reverseNumber(N);
    printf("%d\n",reversed);
    return 0;
}
